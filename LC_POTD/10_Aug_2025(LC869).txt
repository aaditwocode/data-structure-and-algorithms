TOPIC-{BIT MANIPULTION}
class Solution {
public:
    bool reorderedPowerOf2(int n) {
        string s = sig(n);
        for (int i = 0; i < 31; i++) { 
            if (s == sig(1 << i)) return true;
        }
        return false;
    }

private:
    string sig(int x) {
        string s = to_string(x);
        sort(s.begin(), s.end());
        return s;
    }
};
