class Solution
{
    public:
  
    void bubbleSort(int arr[], int n)
    {
        int i, j;
    for (i = 0; i < n - 1; i++){
        for (j = 0; j < n - i - 1; j++){
            if (arr[j] > arr[j + 1])
                swap(arr[j], arr[j + 1]);
}
    }
    
    }
};

compare i to i+1 index if i+1 less than i swap them length of n-1-i as ith elements sorted as of start
tc-o(n^2)
