#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

vector<int> dp;

int rec(int level, int n, int m) {
    if (level > n) return 1e9;       
    if (level == n) return 0;      
    if (dp[level] != -1) return dp[level];

    int ans = 1e9;
    for (int step = 1; step <= m; ++step) {
        int next = level + step;
        if (next <= n) {
            int temp = rec(next, n, m);
            ans = min(ans, 1 + temp);  
        }
    }
    return dp[level] = ans;
}

void solve() {
    int T;
    cin >> T;
    while (T--) {
        int n, m;
        cin >> n >> m;
        dp.assign(n + 1, -1); 
        cout << rec(0, n, m) << endl;
    }
}

int main() {
    solve();
    return 0;
}
